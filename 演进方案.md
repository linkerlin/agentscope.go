## 1. 研究总览

### 1.1 ReMe 项目定位

ReMe (Remember Me, Refine Me) 是 AgentScope 生态中的 **记忆管理框架**，解决 Agent 的两大核心问题：

- **上下文窗口限制**: 长对话中早期信息被截断或丢失
- **会话无状态**: 新会话无法继承历史，总是从头开始

### 1.2 两个项目对比

| 维度 | ReMe (Python) | agentscope.go (当前) |
|------|---------------|---------------------|
| **核心定位** | 记忆管理框架 | Agent 执行框架 |
| **Memory 能力** | 企业级完整方案 | 基础 InMemory |
| **架构模式** | 分层组件化 | 简洁扁平 |
| **持久化** | 文件 + 向量存储 | 无 |
| **压缩/摘要** | ReAct Agent 驱动 | 无 |
| **检索能力** | 向量 + BM25 混合 | 无 |
| **Token 管理** | 精细化计数与裁剪 | 无 |

### 2.2 核心组件详解

#### 2.2.1 ContextChecker - 上下文检查器

```python
# ReMe 实现逻辑
async def check_context(messages, threshold, reserve):
    total_tokens = count_tokens(messages)
    
    if total_tokens <= threshold:
        return [], messages, True  # 无需压缩
    
    # 从尾部保留 reserve tokens
    messages_to_keep = keep_recent(messages, reserve)
    messages_to_compact = messages[:len(messages_to_keep)]
    
    # 完整性校验: 确保 tool_use/tool_result 成对
    is_valid = validate_tool_pairs(messages_to_compact, messages_to_keep)
    
    return messages_to_compact, messages_to_keep, is_valid
```

**关键特性:**
- Token 阈值触发机制
- 保留最近 N tokens 的消息
- 对话完整性保护（不分割 User-Assistant 对）
- Tool 调用对齐检查

#### 2.2.2 Compactor - 智能压缩器

```python
class Compactor:
    """使用 ReAct Agent 压缩对话历史"""
    
    async def compact(self, messages, previous_summary=""):
        # 1. 格式化消息为可读文本
        formatted = format_messages(messages)
        
        # 2. 调用压缩 Agent
        agent = ReActAgent(
            name="reme_compactor",
            sys_prompt=COMPACTOR_PROMPT,  # 指导生成结构化摘要
        )
        
        # 3. 生成结构化摘要
        summary = await agent.reply(formatted)
        
        # 结构化格式:
        # ## Goal
        # ## Constraints
        # ## Progress
        # ## Key Decisions
        # ## Next Steps
        # ## Critical Context
        
        return summary
```

#### 2.2.3 ToolResultCompactor - 工具结果压缩器

```python
class ToolResultCompactor:
    """解决工具输出过长问题"""
    
    async def compact(self, messages, recent_n=1):
        for msg in messages:
            if is_tool_result(msg):
                if is_recent(msg, recent_n):
                    # 最近消息: 保留 100KB，超出部分存文件
                    truncate_to_file(msg, max_bytes=100*1024)
                else:
                    # 旧消息: 激进压缩至 3KB
                    aggressive_truncate(msg, max_bytes=3000)
```

```go
// ============================================================
// 文件: memory/reme_types.go
// ============================================================

package memory

import (
    "context"
    "crypto/sha256"
    "encoding/hex"
    "time"
    
    "github.com/linkerlin/agentscope.go/message"
)

// MemoryType 记忆类型
type MemoryType string

const (
    MemoryTypePersonal   MemoryType = "personal"   // 用户偏好
    MemoryTypeProcedural MemoryType = "procedural" // 任务经验
    MemoryTypeTool       MemoryType = "tool"       // 工具使用
    MemoryTypeSummary    MemoryType = "summary"    // 摘要
)

// MemoryNode 记忆节点 (ReMe Vector-Based 核心)
type MemoryNode struct {
    MemoryID      string                 `json:"memory_id"`
    MemoryType    MemoryType             `json:"memory_type"`
    MemoryTarget  string                 `json:"memory_target"`  // 用户/任务/工具名
    WhenToUse     string                 `json:"when_to_use"`    // 检索条件
    Content       string                 `json:"content"`
    MessageTime   time.Time              `json:"message_time"`
    RefMemoryID   string                 `json:"ref_memory_id"`  // 关联原始记忆
    TimeCreated   time.Time              `json:"time_created"`
    TimeModified  time.Time              `json:"time_modified"`
    Author        string                 `json:"author"`
    Score         float64                `json:"score"`
    Vector        []float32              `json:"vector,omitempty"`
    Metadata      map[string]interface{} `json:"metadata"`
}

// GenerateMemoryID 从内容生成唯一ID
func GenerateMemoryID(content string) string {
    hash := sha256.Sum256([]byte(content))
    return hex.EncodeToString(hash[:])[:16]
}

// NewMemoryNode 创建新记忆节点
func NewMemoryNode(memType MemoryType, target, content string) *MemoryNode {
    now := time.Now()
    return &MemoryNode{
        MemoryID:     GenerateMemoryID(content),
        MemoryType:   memType,
        MemoryTarget: target,
        Content:      content,
        TimeCreated:  now,
        TimeModified: now,
        Metadata:     make(map[string]interface{}),
    }
}

// CompactSummary 结构化压缩摘要
type CompactSummary struct {
    Goal            string   `json:"goal"`
    Constraints     []string `json:"constraints"`
    Progress        string   `json:"progress"`
    KeyDecisions    []string `json:"key_decisions"`
    NextSteps       []string `json:"next_steps"`
    CriticalContext []string `json:"critical_context"`
    Raw             string   `json:"raw"` // 原始文本
}

// ContextCheckResult 上下文检查结果
type ContextCheckResult struct {
    MessagesToCompact []*message.Msg // 需压缩的消息
    MessagesToKeep    []*message.Msg // 保留的消息
    IsValid           bool           // 分割是否有效
    TotalTokens       int            // 总 Token 数
    Threshold         int            // 触发阈值
}

// TokenStats Token 统计
type TokenStats struct {
    TotalMessages         int     `json:"total_messages"`
    CompressedSummaryTokens int   `json:"compressed_summary_tokens"`
    MessagesTokens        int     `json:"messages_tokens"`
    EstimatedTokens       int     `json:"estimated_tokens"`
    MaxInputLength        int     `json:"max_input_length"`
    ContextUsageRatio     float64 `json:"context_usage_ratio"`
}

// RetrieveOptions 检索选项
type RetrieveOptions struct {
    TopK            int                `json:"top_k"`
    MinScore        float64            `json:"min_score"`
    MemoryTypes     []MemoryType       `json:"memory_types,omitempty"`
    MemoryTargets   []string           `json:"memory_targets,omitempty"`
    EnableTimeFilter bool              `json:"enable_time_filter"`
    VectorWeight    float64            `json:"vector_weight"` // 0-1, 0=纯BM25, 1=纯向量
}

// CompactOptions 压缩选项
type CompactOptions struct {
    MaxInputLength     int    `json:"max_input_length"`     // 模型上下文长度
    CompactRatio       float64 `json:"compact_ratio"`       // 触发压缩的阈值比例
    ReserveTokens      int    `json:"reserve_tokens"`       // 保留消息数
    PreviousSummary    string `json:"previous_summary"`     // 之前的摘要(增量更新)
    Language           string `json:"language"`             // 摘要语言
    AddThinkingBlock   bool   `json:"add_thinking_block"`   // 是否添加思考块
}
```

### 4.2 ReMeFileMemory 实现

```go
// ============================================================
// 文件: memory/reme_file_memory.go
// 对应 ReMeLight - 文件-based 轻量记忆
// ============================================================

package memory

import (
    "bufio"
    "context"
    "encoding/json"
    "fmt"
    "os"
    "path/filepath"
    "strings"
    "sync"
    "time"
    
    "github.com/google/uuid"
    "github.com/linkerlin/agentscope.go/agent/react"
    "github.com/linkerlin/agentscope.go/message"
    "github.com/linkerlin/agentscope.go/model"
)

// ReMeFileMemory 文件-based 记忆系统
type ReMeFileMemory struct {
    // 基础存储
    mu       sync.RWMutex
    msgs     []*message.Msg
    msgMarks map[string][]MessageMark // 消息标记 (COMPRESSED, etc.)
    
    // 压缩摘要
    compressedSummary string
    longTermMemory    string
    
    // 路径配置
    workingPath    string
    memoryPath     string
    toolResultPath string
    dialogPath     string
    
    // Token 计数器
    tokenCounter TokenCounter
    
    // 依赖组件
    compactor           *compactor
    summarizer          *summarizer
    toolResultCompactor *toolResultCompactor
    contextChecker      *contextChecker
    msgHandler          *msgHandler
    
    // 配置
    config ReMeFileConfig
}

// MessageMark 消息标记
type MessageMark string

const (
    MarkCompressed MessageMark = "compressed"
    MarkImportant  MessageMark = "important"
    MarkDeleted    MessageMark = "deleted"
)

// ReMeFileConfig 配置
type ReMeFileConfig struct {
    WorkingDir             string        `json:"working_dir"`
    MaxInputLength         int           `json:"max_input_length"`          // 默认 128k
    CompactRatio           float64       `json:"compact_ratio"`             // 默认 0.7
    MemoryCompactReserve   int           `json:"memory_compact_reserve"`    // 默认 10k
    ToolResultRetentionDays int          `json:"tool_result_retention_days"` // 默认 3
    RecentMaxBytes         int           `json:"recent_max_bytes"`          // 默认 100KB
    OldMaxBytes            int           `json:"old_max_bytes"`             // 默认 3KB
    Language               string        `json:"language"`                  // 默认 "zh"
}

// DefaultReMeFileConfig 默认配置
func DefaultReMeFileConfig() ReMeFileConfig {
    return ReMeFileConfig{
        WorkingDir:             ".reme",
        MaxInputLength:         128 * 1024,
        CompactRatio:           0.7,
        MemoryCompactReserve:   10000,
        ToolResultRetentionDays: 3,
        RecentMaxBytes:         100 * 1024,
        OldMaxBytes:            3000,
        Language:               "zh",
    }
}

// NewReMeFileMemory 创建文件-based 记忆系统
func NewReMeFileMemory(config ReMeFileConfig, tokenCounter TokenCounter) (*ReMeFileMemory, error) {
    // 初始化目录结构
    workingPath, _ := filepath.Abs(config.WorkingDir)
    paths := map[string]string{
        "memory":      filepath.Join(workingPath, "memory"),
        "tool_result": filepath.Join(workingPath, "tool_result"),
        "dialog":      filepath.Join(workingPath, "dialog"),
    }
    
    for _, p := range paths {
        if err := os.MkdirAll(p, 0755); err != nil {
            return nil, fmt.Errorf("create directory %s: %w", p, err)
        }
    }
    
    m := &ReMeFileMemory{
        workingPath:    workingPath,
        memoryPath:     paths["memory"],
        toolResultPath: paths["tool_result"],
        dialogPath:     paths["dialog"],
        tokenCounter:   tokenCounter,
        msgMarks:       make(map[string][]MessageMark),
        config:         config,
    }
    
    // 初始化组件
    m.msgHandler = newMsgHandler(tokenCounter)
    m.contextChecker = newContextChecker(tokenCounter)
    m.toolResultCompactor = newToolResultCompactor(m.toolResultPath, config)
    
    return m, nil
}

// Add 添加消息
func (m *ReMeFileMemory) Add(msg *message.Msg) error {
    m.mu.Lock()
    defer m.mu.Unlock()
    
    m.msgs = append(m.msgs, msg)
    return nil
}

// GetAll 获取所有消息 (排除已压缩)
func (m *ReMeFileMemory) GetAll() ([]*message.Msg, error) {
    return m.getMemory(true)
}

// GetRecent 获取最近 N 条
func (m *ReMeFileMemory) GetRecent(n int) ([]*message.Msg, error) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    
    filtered := m.getFilteredMessages()
    if n >= len(filtered) {
        return filtered, nil
    }
    return filtered[len(filtered)-n:], nil
}

// Clear 清空并持久化
func (m *ReMeFileMemory) Clear() error {
    m.mu.Lock()
    defer m.mu.Unlock()
    
    // 持久化所有消息到 dialog
    if err := m.appendToDialog(m.msgs); err != nil {
        return err
    }
    
    m.msgs = nil
    m.msgMarks = make(map[string][]MessageMark)
    return nil
}

// Size 获取消息数量
func (m *ReMeFileMemory) Size() int {
    m.mu.RLock()
    defer m.mu.RUnlock()
    return len(m.getFilteredMessages())
}

// CheckContext 检查上下文大小，决定是否需要压缩
func (m *ReMeFileMemory) CheckContext(ctx context.Context) (*ContextCheckResult, error) {
    m.mu.RLock()
    msgs := m.getFilteredMessages()
    m.mu.RUnlock()
    
    threshold := int(float64(m.config.MaxInputLength) * m.config.CompactRatio * 0.95)
    
    return m.contextChecker.check(ctx, msgs, threshold, m.config.MemoryCompactReserve)
}

// CompactToolResult 压缩工具结果
func (m *ReMeFileMemory) CompactToolResult(ctx context.Context, msgs []*message.Msg, recentN int) ([]*message.Msg, error) {
    return m.toolResultCompactor.compact(ctx, msgs, recentN)
}

// CompactMemory 压缩记忆
func (m *ReMeFileMemory) CompactMemory(ctx context.Context, msgs []*message.Msg, opts CompactOptions) (*CompactSummary, error) {
    if m.compactor == nil {
        return nil, fmt.Errorf("compactor not initialized")
    }
    return m.compactor.compact(ctx, msgs, opts)
}

// SummaryMemory 持久化记忆到文件
func (m *ReMeFileMemory) SummaryMemory(ctx context.Context, msgs []*message.Msg) error {
    if m.summarizer == nil {
        return fmt.Errorf("summarizer not initialized")
    }
    return m.summarizer.summarize(ctx, msgs, m.memoryPath)
}

// PreReasoningHook 推理前钩子 (整合所有预处理)
func (m *ReMeFileMemory) PreReasoningHook(ctx context.Context, msgs []*message.Msg) ([]*message.Msg, *CompactSummary, error) {
    // 1. 压缩工具结果
    msgs, err := m.CompactToolResult(ctx, msgs, 3)
    if err != nil {
        return nil, nil, err
    }
    
    // 2. 检查上下文
    check, err := m.CheckContext(ctx)
    if err != nil {
        return nil, nil, err
    }
    
    if !check.IsValid || len(check.MessagesToCompact) == 0 {
        // 无需压缩
        return msgs, nil, nil
    }
    
    // 3. 压缩记忆
    opts := CompactOptions{
        MaxInputLength:   m.config.MaxInputLength,
        CompactRatio:     m.config.CompactRatio,
        PreviousSummary:  m.compressedSummary,
        Language:         m.config.Language,
        AddThinkingBlock: true,
    }
    
    summary, err := m.CompactMemory(ctx, check.MessagesToCompact, opts)
    if err != nil {
        return nil, nil, err
    }
    
    // 4. 更新内部摘要
    m.mu.Lock()
    m.compressedSummary = summary.Raw
    m.mu.Unlock()
    
    // 5. 异步持久化 (非阻塞)
    go func() {
        ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
        defer cancel()
        _ = m.SummaryMemory(ctx, check.MessagesToCompact)
    }()
    
    // 6. 返回保留的消息 + 摘要
    return check.MessagesToKeep, summary, nil
}

// EstimateTokens 估算 Token 数
func (m *ReMeFileMemory) EstimateTokens() (*TokenStats, error) {
    m.mu.RLock()
    msgs := m.getFilteredMessages()
    summary := m.compressedSummary
    m.mu.RUnlock()
    
    summaryTokens, _ := m.tokenCounter.Count(summary)
    
    var msgTokens int
    for _, msg := range msgs {
        tokens, _ := m.tokenCounter.Count(msg.GetTextContent())
        msgTokens += tokens
    }
    
    estimated := msgTokens + summaryTokens
    
    return &TokenStats{
        TotalMessages:           len(msgs),
        CompressedSummaryTokens: summaryTokens,
        MessagesTokens:          msgTokens,
        EstimatedTokens:         estimated,
        MaxInputLength:          m.config.MaxInputLength,
        ContextUsageRatio:       float64(estimated) / float64(m.config.MaxInputLength) * 100,
    }, nil
}
```

```go
package memory

import "context"

// VectorStore 向量存储接口
type VectorStore interface {
    // 集合管理
    CreateCollection(ctx context.Context, name string, dimension int) error
    DeleteCollection(ctx context.Context, name string) error
    ListCollections(ctx context.Context) ([]string, error)
    
    // CRUD
    Insert(ctx context.Context, nodes []*MemoryNode) error
    Search(ctx context.Context, query string, opts RetrieveOptions) ([]*MemoryNode, error)
    Get(ctx context.Context, memoryID string) (*MemoryNode, error)
    Update(ctx context.Context, node *MemoryNode) error
    Delete(ctx context.Context, memoryID string) error
    DeleteAll(ctx context.Context) error
    
    // 列表查询
    List(ctx context.Context, filters map[string]interface{}, limit int) ([]*MemoryNode, error)
}

// EmbeddingModel 嵌入模型接口
type EmbeddingModel interface {
    Embed(ctx context.Context, text string) ([]float32, error)
    EmbedBatch(ctx context.Context, texts []string) ([][]float32, error)
}

// LocalVectorStore 本地向量存储 (基于 HNSW 或简单线性搜索)
type LocalVectorStore struct {
    collection string
    dimension  int
    embedding  EmbeddingModel
    
    mu    sync.RWMutex
    nodes map[string]*MemoryNode // memoryID -> node
}

func NewLocalVectorStore(embedding EmbeddingModel) *LocalVectorStore {
    return &LocalVectorStore{
        embedding: embedding,
        nodes:     make(map[string]*MemoryNode),
    }
}

func (s *LocalVectorStore) Insert(ctx context.Context, nodes []*MemoryNode) error {
    // 生成向量
    for _, node := range nodes {
        if len(node.Vector) == 0 {
            vector, err := s.embedding.Embed(ctx, node.Content)
            if err != nil {
                return err
            }
            node.Vector = vector
        }
    }
    
    s.mu.Lock()
    defer s.mu.Unlock()
    
    for _, node := range nodes {
        s.nodes[node.MemoryID] = node
    }
    return nil
}

func (s *LocalVectorStore) Search(ctx context.Context, query string, opts RetrieveOptions) ([]*MemoryNode, error) {
    // 1. 查询向量化
    queryVec, err := s.embedding.Embed(ctx, query)
    if err != nil {
        return nil, err
    }
    
    s.mu.RLock()
    defer s.mu.RUnlock()
    
    // 2. 计算相似度 (余弦相似度)
    type scoredNode struct {
        node  *MemoryNode
        score float64
    }
    
    var scored []scoredNode
    for _, node := range s.nodes {
        // 过滤器
        if !matchesFilter(node, opts) {
            continue
        }
        
        score := cosineSimilarity(queryVec, node.Vector)
        if score >= opts.MinScore {
            scored = append(scored, scoredNode{node, score})
        }
    }
    
    // 3. 排序并返回 TopK
    sort.Slice(scored, func(i, j int) bool {
        return scored[i].score > scored[j].score
    })
    
    topK := opts.TopK
    if topK > len(scored) {
        topK = len(scored)
    }
    
    result := make([]*MemoryNode, topK)
    for i := 0; i < topK; i++ {
        result[i] = scored[i].node
        result[i].Score = scored[i].score
    }
    
    return result, nil
}

func cosineSimilarity(a, b []float32) float64 {
    if len(a) != len(b) {
        return 0
    }
    
    var dot, normA, normB float64
    for i := 0; i < len(a); i++ {
        dot += float64(a[i]) * float64(b[i])
        normA += float64(a[i]) * float64(a[i])
        normB += float64(b[i]) * float64(b[i])
    }
    
    if normA == 0 || normB == 0 {
        return 0
    }
    
    return dot / (math.Sqrt(normA) * math.Sqrt(normB))
}

func matchesFilter(node *MemoryNode, opts RetrieveOptions) bool {
    if len(opts.MemoryTypes) > 0 {
        found := false
        for _, t := range opts.MemoryTypes {
            if node.MemoryType == t {
                found = true
                break
            }
        }
        if !found {
            return false
        }
    }
    
    if len(opts.MemoryTargets) > 0 {
        found := false
        for _, t := range opts.MemoryTargets {
            if node.MemoryTarget == t {
                found = true
                break
            }
        }
        if !found {
            return false
        }
    }
    
    return true
}
```

```go
package memory

import (
    "context"
```

```go
    "strings"
    
    "github.com/linkerlin/agentscope.go/agent/react"

// compactor 使用 ReAct Agent 压缩对话历史
type compactor struct {
    model       model.ChatModel
    tokenCounter TokenCounter
}

func newCompactor(m model.ChatModel, tc TokenCounter) *compactor {
    return &compactor{
        model:        m,
        tokenCounter: tc,
    }
}

const compactorSystemPrompt = `You are a memory compactor. Your task is to compress conversation history into a structured summary.

The summary must follow this exact format:

## Goal
[User's goals and objectives]

## Constraints
[Any constraints or preferences mentioned]

## Progress
[What has been accomplished so far]

## Key Decisions
[Important decisions made during the conversation]

## Next Steps
[Planned next actions]

## Critical Context
[Important data like file paths, function names, error messages]

Rules:
1. Be concise but preserve all critical information
2. Maintain continuity for the next conversation turn
3. Use bullet points where appropriate
4. If previous summary is provided, merge new information intelligently
5. Output ONLY the formatted summary, no extra text`

func (c *compactor) compact(ctx context.Context, msgs []*message.Msg, opts CompactOptions) (*CompactSummary, error) {
    // 格式化消息
    formatted := c.formatMessages(msgs)
    
    // 构建用户提示
    var userPrompt strings.Builder
    userPrompt.WriteString("# Conversation\n")
    userPrompt.WriteString(formatted)
    userPrompt.WriteString("\n\n")
    
    if opts.PreviousSummary != "" {
        userPrompt.WriteString("# Previous Summary\n")
        userPrompt.WriteString(opts.PreviousSummary)
        userPrompt.WriteString("\n\n")
        userPrompt.WriteString("Update the summary with new information from the conversation.")
    } else {
        userPrompt.WriteString("Create a summary of this conversation.")
    }
    
    if opts.Language == "zh" {
        userPrompt.WriteString("\n\n请用中文输出。")
    }
    
    // 创建 Agent
    agent, err := react.Builder().
        Name("reme_compactor").
        SysPrompt(compactorSystemPrompt).
        Model(c.model).
        Build()
    if err != nil {
        return nil, err
    }
    
    // 执行压缩
    resp, err := agent.Call(ctx, message.NewMsg().
        Role(message.RoleUser).
        TextContent(userPrompt.String()).
        Build())
    if err != nil {
        return nil, err
    }
    
    raw := resp.GetTextContent()
    
    // 解析结构化摘要
    summary := c.parseSummary(raw)
    summary.Raw = raw
    
    return summary, nil
}

func (c *compactor) formatMessages(msgs []*message.Msg) string {
    var parts []string
    for _, msg := range msgs {
        role := msg.Role
        content := msg.GetTextContent()
        parts = append(parts, fmt.Sprintf("[%s]: %s", role, content))
    }
    return strings.Join(parts, "\n\n")
}

func (c *compactor) parseSummary(raw string) *CompactSummary {
    s := &CompactSummary{}
    
    lines := strings.Split(raw, "\n")
    var currentSection string
    var currentContent []string
    
    flushSection := func() {
        content := strings.TrimSpace(strings.Join(currentContent, "\n"))
        switch currentSection {
        case "Goal":
            s.Goal = content
        case "Constraints":
            s.Constraints = strings.Split(content, "\n")
        case "Progress":
            s.Progress = content
        case "Key Decisions":
            s.KeyDecisions = strings.Split(content, "\n")
        case "Next Steps":
            s.NextSteps = strings.Split(content, "\n")
        case "Critical Context":
            s.CriticalContext = strings.Split(content, "\n")
        }
        currentContent = nil
    }
    
    for _, line := range lines {
        line = strings.TrimSpace(line)
        if strings.HasPrefix(line, "## ") {
            flushSection()
            currentSection = strings.TrimPrefix(line, "## ")
        } else if currentSection != "" {
            currentContent = append(currentContent, line)
        }
    }
    flushSection()
    
    return s
}
```

```go
func main() {
    ctx := context.Background()
    
    // 1. 创建 Token 计数器
    tokenCounter := memory.NewSimpleTokenCounter()
    
    // 2. 创建 ReMeFileMemory
    memConfig := memory.DefaultReMeFileConfig()
    memConfig.WorkingDir = ".my_agent_memory"
    memConfig.MaxInputLength = 128000
    
    mem, err := memory.NewReMeFileMemory(memConfig, tokenCounter)
    if err != nil {
        log.Fatal(err)
    }
    
    // 3. 创建模型 (用于压缩)
    model, _ := openai.Builder().
        APIKey("your-api-key").
        ModelName("gpt-4o-mini").
        Build()

        // ... 更多消息

    for _, msg := range messages {
        mem.Add(msg)
    }
    
    // 7. 推理前处理 (自动压缩)
    processedMsgs, summary, err := mem.PreReasoningHook(ctx, messages)

    // 8. 使用处理后的消息调用 Agent
    agent, _ := react.Builder().
        Name("Assistant").
        Model(model).
        Memory(mem).  // 使用 ReMeMemory

    resp, _ := agent.Call(ctx, message.NewMsg().
        Role(message.RoleUser).
        TextContent("给我推荐一些学习资源").
        Build())
    
    fmt.Println(resp.GetTextContent())
```

## 8. 总结

### 8.1 整合价值总结

| 能力 | ReMe 引入前 | ReMe 引入后 |
|------|------------|------------|
| **上下文管理** | 无限制增长 | Token 感知，自动压缩 |
| **工具结果处理** | 直接存储 | 智能压缩，文件引用 |
| **长期记忆** | 无 | 文件/Markdown 存储 |
| **语义检索** | 无 | 向量检索 + BM25 |
| **个人化** | 无 | 用户偏好记忆 |
| **经验积累** | 无 | 程序记忆、工具记忆 |

### 8.2 技术要点

1. **Go 惯用设计**: 使用 interface、struct embedding、context 传递
2. **向后兼容**: 基础 Memory 接口保持不变，新功能通过扩展接口提供
3. **模块化**: File-based 和 Vector-based 可独立使用或组合
4. **可配置**: 通过 Config 结构体灵活配置各项参数
5. **异步友好**: 压缩、摘要等操作支持异步执行

### 8.3 与 AgentScope 生态对齐

```
AgentScope-Java ────► 企业级完整方案
      │
      ▼
  ReMe (Python) ────► Memory 专业框架
      │
      ▼
agentscope.go ────► 轻量 Go 实现 + ReMe 整合
```

通过本方案，agentscope.go 将获得业界领先的 Memory 管理能力，同时保持 Go 语言的简洁和高效特性。
